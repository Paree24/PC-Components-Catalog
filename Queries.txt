Complex SQL Queries
_____________________

SELECT m.Mname 
FROM motherboard m, cpu c 
WHERE m.Slot=c.Slot 
AND c.Cname=[Name from POST METHOD]

Example Input for Above : SELECT m.Mname FROM motherboard m, cpu c WHERE m.Slot=c.Slot AND c.Cname="Ryzen 3 1300X" 

SELECT r.Brand,r.ClockSpeed,r.Size,r.RAMSlot,r.Price 
FROM motherboard m, ram r 
WHERE m.RAMSlot=r.RAMSlot 
AND m.Mname=[Name from POST METHOD]

Example : SELECT r.Brand,r.ClockSpeed,r.Size,r.RAMSlot,r.Price FROM motherboard m, ram r WHERE m.RAMSlot=r.RAMSlot AND m.Slot="AM4" 

SELECT c.Cname,c.Brand,c.Threads,c.Cores,c.Price,c.Performance,c.Performance/c.Price AS 'Performance for price' 
FROM cpu c  
ORDER BY `Performance for price`  
DESC

SELECT g.Gname,g.Brand,g.VRAMType,g.VRAMCapacity,g.Price,g.Performance, g.Performance/g.Price AS 'Performance for price' 
FROM gpu g 
ORDER BY `Performance for price` 
DESC
